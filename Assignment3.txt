create table Student(
    snum integer,
    sname varchar(30),
    major varchar(30),
    lvl varchar(30),
    age integer, primary key(snum));

create table Faculty(
    fid integer,
    fname varchar(30),
    deptid integer,
    primary key(fid));

create table Class(
    cname varchar(30),
    meetat timestamp,
    room varchar(30),
    fid integer,
    primary key(cname),
    foreign key(fid) references Faculty(fid));


create table Enrolled(
    snum integer,
    cname varchar(30),
    primary key(snum),
    foreign key(snum) references Student(snum),
    foreign key(cname) references Class(cname));
